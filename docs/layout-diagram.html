const BOARD = 85, PAD = 52;


const C = [

const FL = [

  ['cup','hop','items dump'],['cup','wdg','String A'],

  ['wdg','whl','golf ball'],['whl','inc','trigger'],

  ['inc','scr','100g falls'],['scr','hop','gate pin'],

  ['hop','srt','items flow'],['srt','bar','non-magnetic'],

  ['mag','c2','clips'],['bar','c1','marbles'],['bar','c3','tees']

];


let showM=true, showL=true, showF=false, view='both', hov=null, animId=null;

const tip = document.getElementById('tip');


function px(cm, size) { return PAD + (cm / BOARD) * (size - PAD * 2); }


function grid(ctx, w, h) {

  ctx.strokeStyle = 'rgba(233,69,96,.05)';

  ctx.lineWidth = 1;

  for (let i = 0; i <= BOARD; i += 5) {

    let a = px(i, w);

    ctx.beginPath(); ctx.moveTo(a, PAD); ctx.lineTo(a, h - PAD); ctx.stroke();

    a = px(i, h);

    ctx.beginPath(); ctx.moveTo(PAD, a); ctx.lineTo(w - PAD, a); ctx.stroke();

  }

}


function axes(ctx, w, h, xl, yl) {

  ctx.strokeStyle = 'rgba(255,255,255,.15)';

  ctx.lineWidth = 1;

  ctx.beginPath(); ctx.moveTo(PAD, PAD); ctx.lineTo(PAD, h - PAD); ctx.lineTo(w - PAD, h - PAD); ctx.stroke();

  ctx.fillStyle = '#7d8590';

  ctx.font = '10px "DM Mono", monospace';

  for (let i = 0; i <= BOARD; i += 10) {

    ctx.fillText(i, px(i, w) - 6, h - PAD + 16);

    const yy = h - PAD - ((i / BOARD) * (h - PAD * 2));

    ctx.fillText(i, PAD - 24, yy + 4);

  }

  ctx.fillStyle = '#e94560';

  ctx.font = '11px "DM Mono", monospace';

  ctx.fillText(xl, w / 2 - 40, h - PAD + 36);

  ctx.save();

  ctx.translate(14, h / 2 + 30);

  ctx.rotate(-Math.PI / 2);

  ctx.fillText(yl, 0, 0);

  ctx.restore();

}


function box(ctx, c, key, w, h, side) {

  const v = c[key];

  if (!v) return;

  const sc = (w - PAD * 2) / BOARD;

  let bx, by, bw, bh;

  if (side) {

    bx = PAD + v.x * sc;

    by = h - PAD - (v.y + v.h) * sc;

    bw = v.w * sc;

    bh = v.h * sc;

  } else {

    bx = PAD + v.x * sc;

    by = PAD + v.y * sc;

    bw = v.w * sc;

    bh = v.h * sc;

  }

  const hi = hov === c.id;


  ctx.fillStyle = c.col + (hi ? 'bb' : '33');

  ctx.strokeStyle = c.col + (hi ? 'ff' : '77');

  ctx.lineWidth = hi ? 2.5 : 1;

  ctx.fillRect(bx, by, bw, bh);

  ctx.strokeRect(bx, by, bw, bh);


  if (showL) {

    ctx.fillStyle = hi ? '#fff' : c.col;

    ctx.font = (hi ? 'bold ' : '') + '9px "DM Mono", monospace';

    const lb = c.n.length > 22 ? c.n.slice(0, 20) + '..' : c.n;

    ctx.fillText(lb, bx + 3, by + 12);

  }

  if (showM) {

    ctx.fillStyle = '#7d8590';

    ctx.font = '8px "DM Mono", monospace';

    if (side) {

      ctx.fillText(v.w + '\u00d7' + v.h + ' cm', bx + 3, by + bh - 4);

      if (bh > 22) ctx.fillText('h=' + v.y + '\u2013' + (v.y + v.h), bx + 3, by + bh - 14);

    } else {

      ctx.fillText(v.w + '\u00d7' + v.h + ' cm', bx + 3, by + bh - 4);

    }

  }

  c['_r' + key] = { x: bx, y: by, w: bw, h: bh };

}


function arrows(ctx, key) {

  if (!showF) return;

  const t = performance.now() / 1000;

  FL.forEach(([fid, tid, lab], i) => {

    const fc = C.find(c => c.id === fid);

    const tc = C.find(c => c.id === tid);

    if (!fc || !tc) return;

    const r1 = fc['_r' + key], r2 = tc['_r' + key];

    if (!r1 || !r2) return;

    const x1 = r1.x + r1.w / 2, y1 = r1.y + r1.h / 2;

    const x2 = r2.x + r2.w / 2, y2 = r2.y + r2.h / 2;


    ctx.strokeStyle = '#f77f00aa';

    ctx.lineWidth = 1.5;

    ctx.setLineDash([6, 4]);

    ctx.lineDashOffset = -(t * 30 + i * 20) % 20;

    ctx.beginPath(); ctx.moveTo(x1, y1); ctx.lineTo(x2, y2); ctx.stroke();

    ctx.setLineDash([]);


    const a = Math.atan2(y2 - y1, x2 - x1);

    const ax = x2 - Math.cos(a) * 8, ay = y2 - Math.sin(a) * 8;

    ctx.fillStyle = '#f77f00';

    ctx.beginPath(); ctx.moveTo(x2, y2);

    ctx.lineTo(ax - Math.sin(a) * 4, ay + Math.cos(a) * 4);

    ctx.lineTo(ax + Math.sin(a) * 4, ay - Math.cos(a) * 4);

    ctx.fill();


    ctx.fillStyle = '#f77f00';

    ctx.font = '8px "DM Mono", monospace';

    ctx.fillText(lab, (x1 + x2) / 2 + 4, (y1 + y2) / 2 - 4);

  });

}


const sC = document.getElementById('S'), tC = document.getElementById('T');

const sP = document.getElementById('sidePanel'), tP = document.getElementById('topPanel');


function resize() {

  const single = view !== 'both';

  const cw = single

    ? Math.min(900, window.innerWidth - 60)

    : Math.min(580, Math.floor((window.innerWidth - 64) / 2));

  const ch = single

    ? Math.min(780, window.innerHeight - 200)

    : Math.min(580, window.innerHeight - 220);

  sC.width = cw; sC.height = ch;

  tC.width = cw; tC.height = ch;

  sP.style.display = (view === 'top') ? 'none' : 'block';

  tP.style.display = (view === 'side') ? 'none' : 'block';

}


function draw() {

  resize();

  if (view !== 'top') {

    const ctx = sC.getContext('2d'), w = sC.width, h = sC.height;

    ctx.clearRect(0, 0, w, h);

    grid(ctx, w, h);

    axes(ctx, w, h, 'x position (cm)', 'height (cm)');

    C.forEach(c => box(ctx, c, 's', w, h, true));

    arrows(ctx, 's');

  }

  if (view !== 'side') {

    const ctx = tC.getContext('2d'), w = tC.width, h = tC.height;

    ctx.clearRect(0, 0, w, h);

    grid(ctx, w, h);

    axes(ctx, w, h, 'x position (cm)', 'y depth (cm)');

    C.forEach(c => box(ctx, c, 't', w, h, false));

    arrows(ctx, 't');

  }

  if (showF) animId = requestAnimationFrame(draw);

}


document.querySelectorAll('[data-v]').forEach(b => b.addEventListener('click', () => {

  document.querySelectorAll('[data-v]').forEach(x => x.classList.remove('on'));

  b.classList.add('on');

  view = b.dataset.v;

  draw();

}));


document.querySelectorAll('[data-t]').forEach(b => b.addEventListener('click', () => {

  b.classList.toggle('on');

  const k = b.dataset.t;

  if (k === 'meas') showM = !showM;

  if (k === 'lbl')  showL = !showL;

  if (k === 'flow') {

    showF = !showF;

    if (showF) requestAnimationFrame(draw);

    else if (animId) { cancelAnimationFrame(animId); animId = null; }

  }

  draw();

}));


function hit(e, cv, key) {

  const r = cv.getBoundingClientRect();

  const mx = e.clientX - r.left, my = e.clientY - r.top;

  let f = null;

  C.forEach(c => {

    const b = c['_r' + key];

    if (b && mx >= b.x && mx <= b.x + b.w && my >= b.y && my <= b.y + b.h) f = c;

  });

  if (f) {

    hov = f.id;

    tip.innerHTML = '<b>' + f.n + '</b>' + f.d;

    tip.style.display = 'block';

    tip.style.left = (e.clientX + 14) + 'px';

    tip.style.top = (e.clientY - 8) + 'px';

  } else {

    hov = null;

    tip.style.display = 'none';

  }

  draw();

}


sC.addEventListener('mousemove', e => hit(e, sC, 's'));

tC.addEventListener('mousemove', e => hit(e, tC, 't'));

[sC, tC].forEach(c => c.addEventListener('mouseleave', () => {

  hov = null; tip.style.display = 'none'; draw();

}));

window.addEventListener('resize', draw);

draw();
