let currentStage = 0;

let playing = false;

let stageProgress = 0;

let lastTime = 0;

const STAGE_DURATION = 3500;


const cv = document.getElementById('cv');

const ctx = cv.getContext('2d');

const stageRow = document.getElementById('stageRow');


stages.forEach((s, i) => {

  const pip = document.createElement('div');

  pip.className = 'stage-pip' + (i === 0 ? ' current' : '');

  pip.textContent = s.id;

  pip.addEventListener('click', () => goToStage(i));

  stageRow.appendChild(pip);

});


function updateUI() {

  const s = stages[currentStage];

  document.getElementById('stepLabel').textContent =

    'Step ' + (currentStage + 1) + ' / ' + stages.length;

  document.getElementById('progFill').style.width =

    ((currentStage + stageProgress) / stages.length * 100) + '%';


  document.querySelectorAll('.stage-pip').forEach((p, i) => {

    p.className = 'stage-pip' +

      (i < currentStage ? ' done' : '') +

      (i === currentStage ? ' current' : '');

  });


  document.getElementById('cardTitle').textContent = s.name;

  document.getElementById('cardDesc').textContent = s.desc;

  document.getElementById('cardDetail').textContent = s.detail;

  document.getElementById('physDesc').textContent = s.phys;

  document.getElementById('physDetail').textContent = s.physDetail;


  document.getElementById('btnPlay').innerHTML =

    playing ? '&#10074;&#10074; Pause' : '&#9654; Play';

  document.getElementById('btnPlay').classList.toggle('active', playing);

}


function goToStage(i) {

  currentStage = Math.max(0, Math.min(stages.length - 1, i));

  stageProgress = 0;

  updateUI();

  drawScene();

}


document.getElementById('btnPrev').addEventListener('click', () => goToStage(currentStage - 1));

document.getElementById('btnNext').addEventListener('click', () => goToStage(currentStage + 1));

document.getElementById('btnReset').addEventListener('click', () => { playing = false; goToStage(0); });

document.getElementById('btnPlay').addEventListener('click', () => {

  playing = !playing;

  if (playing) { lastTime = performance.now(); requestAnimationFrame(animate); }

  updateUI();

});


document.getElementById('progBar').addEventListener('click', e => {

  const r = e.target.getBoundingClientRect();

  const pct = (e.clientX - r.left) / r.width;

  const idx = Math.floor(pct * stages.length);

  goToStage(idx);

});


function animate(now) {

  if (!playing) return;

  const dt = now - lastTime;

  lastTime = now;

  stageProgress += dt / STAGE_DURATION;

  if (stageProgress >= 1) {

    stageProgress = 0;

    currentStage++;

    if (currentStage >= stages.length) {

      currentStage = stages.length - 1;

      stageProgress = 1;

      playing = false;

    }

  }

  updateUI();

  drawScene();

  if (playing) requestAnimationFrame(animate);

}


function lerp(a, b, t) { return a + (b - a) * t; }

function ease(t) { return t < .5 ? 2*t*t : -1+(4-2*t)*t; }


function drawScene() {

  const W = cv.width, H = cv.height;

  ctx.clearRect(0, 0, W, H);


  const t = ease(stageProgress);

  const active = currentStage;


  ctx.fillStyle = '#141820';

  ctx.fillRect(0, 0, W, H);


  drawBoard(W, H);


  drawCupHopper(active, t, W, H);

  drawWedge(active, t, W, H);

  drawWheelAxle(active, t, W, H);

  drawIncline(active, t, W, H);

  drawScrew(active, t, W, H);

  drawSortRamp(active, t, W, H);

  drawBarGrid(active, t, W, H);

  drawContainers(active, t, W, H);


  if (active >= 0) drawEnergyPulse(active, t, W, H);

}


function drawBoard(W, H) {

  ctx.strokeStyle = '#2a3040';

  ctx.lineWidth = 2;

  ctx.strokeRect(40, 30, W - 80, H - 60);


  ctx.fillStyle = '#1a1e28';

  ctx.fillRect(40, H - 30, W - 80, 2);


  ctx.fillStyle = '#2a3040';

  ctx.font = '10px "IBM Plex Mono"';

  ctx.fillText('85 cm', W / 2 - 15, H - 12);

  ctx.save();

  ctx.translate(18, H / 2);

  ctx.rotate(-Math.PI / 2);

  ctx.fillText('85 cm height', 0, 0);

  ctx.restore();

}


function glow(color, blur) {

  ctx.shadowColor = color;

  ctx.shadowBlur = blur;

}

function noGlow() {

  ctx.shadowColor = 'transparent';

  ctx.shadowBlur = 0;

}


function drawCupHopper(active, t, W, H) {

  const isActive = active === 0;

  const isDone = active > 0;

  const cupX = 100, cupBaseY = 120;

  const cupW = 50, cupH = 35;


  let angle = 0;

  if (isActive) angle = t * 1.2;

  else if (isDone) angle = 1.2;


  ctx.save();

  ctx.translate(cupX, cupBaseY);

  ctx.rotate(angle);


  if (isActive) glow('#fc6681', 12);

  ctx.fillStyle = isDone ? '#3a2030' : '#fc668188';

  ctx.strokeStyle = isActive ? '#fc6681' : (isDone ? '#5a3040' : '#fc668166');

  ctx.lineWidth = isActive ? 2 : 1;

  ctx.beginPath();

  ctx.moveTo(-cupW/2, 0);

  ctx.lineTo(-cupW/2 + 8, -cupH);

  ctx.lineTo(cupW/2 - 8, -cupH);

  ctx.lineTo(cupW/2, 0);

  ctx.closePath();

  ctx.fill(); ctx.stroke();

  noGlow();


  if (isActive && t > 0.2) {

    const count = Math.floor((t - 0.2) * 12);

    for (let i = 0; i < count; i++) {

      const px = cupX + Math.sin(i * 2.3) * 15;

      const py = cupBaseY + 20 + i * 8 + (t - 0.2) * 60;

      const colors = ['#fc6681', '#fcdb66', '#66fcf1'];

      ctx.fillStyle = colors[i % 3];

      ctx.beginPath();

      ctx.arc(px, Math.min(py, 380), 3, 0, Math.PI * 2);

      ctx.fill();

    }

  }

  ctx.restore();


  ctx.fillStyle = isActive ? '#fc6681' : '#5a3040';

  ctx.font = (isActive ? '600 ' : '') + '10px "IBM Plex Mono"';

  ctx.fillText('CUP', cupX - 12, cupBaseY - cupH - 10);


  const hopX = 90, hopY = 200, hopW = 70, hopH = 160;

  const gateOpen = active >= 4 ? 1 : (active === 4 ? t : 0);


  ctx.fillStyle = '#fc668122';

  ctx.strokeStyle = active >= 4 ? '#45a29e' : '#fc668155';

  ctx.lineWidth = 1;

  ctx.beginPath();

  ctx.moveTo(hopX, hopY);

  ctx.lineTo(hopX - 15, hopY + hopH);

  ctx.lineTo(hopX + hopW + 15, hopY + hopH);

  ctx.lineTo(hopX + hopW, hopY);

  ctx.closePath();

  ctx.fill(); ctx.stroke();


  if (active < 5) {

    ctx.fillStyle = '#fc668188';

    ctx.fillRect(hopX + 5, hopY + hopH - 5, hopW - 10, 5);

  }


  ctx.fillStyle = '#5a3040';

  ctx.font = '10px "IBM Plex Mono"';

  ctx.fillText('HOPPER', hopX + 8, hopY + 20);

}


function drawWedge(active, t, W, H) {

  const isActive = active === 1;

  const isDone = active > 1;

  const baseX = 260, baseY = 115;


  if (isActive) glow('#fcdb66', 10);

  ctx.strokeStyle = isActive ? '#fcdb66' : (isDone ? '#5a4a20' : '#fcdb6644');

  ctx.fillStyle = isActive ? '#fcdb6633' : '#fcdb6611';

  ctx.lineWidth = isActive ? 2 : 1;

  ctx.beginPath();

  ctx.moveTo(baseX, baseY);

  ctx.lineTo(baseX + 120, baseY + 30);

  ctx.lineTo(baseX, baseY + 30);

  ctx.closePath();

  ctx.fill(); ctx.stroke();

  noGlow();


  let wedgeOffset = 0;

  if (isActive) wedgeOffset = t * 30;

  else if (isDone) wedgeOffset = 30;


  ctx.fillStyle = isActive ? '#fcdb66' : '#5a4a20';

  ctx.beginPath();

  ctx.moveTo(baseX + 10 - wedgeOffset, baseY + 5);

  ctx.lineTo(baseX + 25 - wedgeOffset, baseY + 5);

  ctx.lineTo(baseX + 10 - wedgeOffset, baseY + 25);

  ctx.closePath();

  ctx.fill();


  if (isActive && t > 0.3) {

    const ballX = lerp(baseX + 30, baseX + 120, (t - 0.3) / 0.7);

    const ballY = lerp(baseY + 10, baseY + 25, (t - 0.3) / 0.7);

    glow('#fcdb66', 8);

    ctx.fillStyle = '#fcdb66';

    ctx.beginPath();

    ctx.arc(ballX, ballY, 6, 0, Math.PI * 2);

    ctx.fill();

    noGlow();

  } else if (isDone) {

    ctx.fillStyle = '#5a4a20';

    ctx.beginPath();

    ctx.arc(baseX + 120, baseY + 25, 6, 0, Math.PI * 2);

    ctx.fill();

  } else {

    ctx.fillStyle = '#fcdb6644';

    ctx.beginPath();

    ctx.arc(baseX + 30, baseY + 10, 6, 0, Math.PI * 2);

    ctx.fill();

  }


  ctx.fillStyle = isActive ? '#fcdb66' : '#5a4a20';

  ctx.font = (isActive ? '600 ' : '') + '10px "IBM Plex Mono"';

  ctx.fillText('1. WEDGE', baseX, baseY - 8);


  if (active >= 1) {

    ctx.strokeStyle = '#f77f0044';

    ctx.setLineDash([4, 4]);

    ctx.beginPath();

    ctx.moveTo(145, 130);

    ctx.lineTo(baseX - 5, baseY + 15);

    ctx.stroke();

    ctx.setLineDash([]);

    ctx.fillStyle = '#f77f00';

    ctx.font = '8px "IBM Plex Mono"';

    ctx.fillText('String A', 170, 125);

  }

}


function drawWheelAxle(active, t, W, H) {

  const isActive = active === 2;

  const isDone = active > 2;

  const cx = 500, cy = 200;

  const wheelR = 36, axleR = 18;


  let rotation = 0;

  if (isActive) rotation = t * Math.PI * 2;

  else if (isDone) rotation = Math.PI * 2;


  if (isActive) glow('#fcdb66', 10);

  ctx.strokeStyle = isActive ? '#fcdb66' : (isDone ? '#5a4a20' : '#fcdb6644');

  ctx.lineWidth = isActive ? 2 : 1;

  ctx.beginPath();

  ctx.arc(cx, cy, wheelR, 0, Math.PI * 2);

  ctx.stroke();

  ctx.beginPath();

  ctx.arc(cx, cy, axleR, 0, Math.PI * 2);

  ctx.stroke();

  noGlow();


  ctx.save();

  ctx.translate(cx, cy);

  ctx.rotate(rotation);

  ctx.strokeStyle = isActive ? '#fcdb6688' : '#fcdb6622';

  for (let i = 0; i < 4; i++) {

    ctx.beginPath();

    ctx.moveTo(0, 0);

    ctx.lineTo(wheelR * Math.cos(i * Math.PI / 2), wheelR * Math.sin(i * Math.PI / 2));

    ctx.stroke();

  }

  ctx.restore();


  const bucketY = isActive ? lerp(cy - wheelR - 20, cy + wheelR + 40, t) : (isDone ? cy + wheelR + 40 : cy - wheelR - 20);

  ctx.fillStyle = isActive ? '#fcdb6688' : '#fcdb6633';

  ctx.fillRect(cx + wheelR + 10, bucketY, 20, 15);

  ctx.strokeStyle = isActive ? '#fcdb66' : '#fcdb6644';

  ctx.strokeRect(cx + wheelR + 10, bucketY, 20, 15);


  const weightY = isActive ? lerp(cy + axleR + 30, cy - axleR - 10, t) : (isDone ? cy - axleR - 10 : cy + axleR + 30);

  ctx.fillStyle = isActive ? '#fc668188' : '#fc668133';

  ctx.fillRect(cx - axleR - 30, weightY, 16, 16);

  ctx.strokeStyle = isActive ? '#fc6681' : '#fc668144';

  ctx.strokeRect(cx - axleR - 30, weightY, 16, 16);


  ctx.fillStyle = isActive ? '#fcdb66' : '#5a4a20';

  ctx.font = (isActive ? '600 ' : '') + '10px "IBM Plex Mono"';

  ctx.fillText('2. WHEEL & AXLE', cx - 50, cy - wheelR - 16);


  if (isActive) {

    ctx.fillStyle = '#fcdb66';

    ctx.font = '9px "IBM Plex Mono"';

    ctx.fillText('IMA = 2.0', cx - 18, cy + 4);

    ctx.fillText('R=6cm', cx + wheelR + 4, cy - 8);

    ctx.fillText('r=3cm', cx + axleR + 4, cy + 12);

  }

}


function drawIncline(active, t, W, H) {

  const isActive = active === 3;

  const isDone = active > 3;

  const x1 = 620, y1 = 290, x2 = 820, y2 = 220;


  if (isActive) glow('#fcdb66', 10);

  ctx.strokeStyle = isActive ? '#fcdb66' : (isDone ? '#5a4a20' : '#fcdb6644');

  ctx.fillStyle = isActive ? '#fcdb6622' : '#fcdb6611';

  ctx.lineWidth = isActive ? 2 : 1;

  ctx.beginPath();

  ctx.moveTo(x1, y1);

  ctx.lineTo(x2, y2);

  ctx.lineTo(x2, y1);

  ctx.closePath();

  ctx.fill(); ctx.stroke();

  noGlow();


  let objPos = 0;

  if (isActive) objPos = t;

  else if (isDone) objPos = 1;


  const objX = lerp(x1 + 10, x2 - 20, objPos);
